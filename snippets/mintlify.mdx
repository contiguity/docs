<img 
    src="https://fake.img.com/nonexistent.jpg" 
    onError={() => {
        const removeLine = () => {
            document.querySelector('a[href*="https://mintlify.com/preview-request"]')?.remove();
            document.querySelector('#nav-separator-line')?.remove();
            const nav = document.querySelector("#navigation-items");
            if (nav?.lastElementChild) {
                const line = document.createElement("div");
                Object.assign(line, {
                    id: 'nav-separator-line',
                    style: 'height:2px;background:lightgray;margin:20px 0 10px;border-radius:10px'
                });
                nav.insertBefore(line, nav.lastElementChild);
            }
        };
        removeLine();
        document.addEventListener('routeChange', removeLine);
    }}
    style={{display: 'none'}}
